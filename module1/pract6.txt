Combining and appending datasets using MERGE in SAS, Merge Files in SPSS,
and merge() or bind_rows() in R.


# Install required package (run once)
install.packages("dplyr")

# Load the package
library(dplyr)

# --- Creating first dataset ---
data1 <- data.frame(
  ID = c(1, 2, 3),
  Name = c("Alice", "John", "Rahul")
)

# --- Creating second dataset ---
data2 <- data.frame(
  ID = c(2, 3, 4),
  Score = c(90, 85, 92)
)

# --- Merging datasets based on common column (ID) ---
merged_data <- merge(data1, data2, by = "ID", all = TRUE)

# --- Appending datasets using bind_rows() ---
# (Adding new dataset with same structure)
data3 <- data.frame(
  ID = c(5, 6),
  Name = c("Meera", "Rohit"),
  Score = c(88, 91)
)

# Append data
appended_data <- bind_rows(merged_data, data3)

# Display final result
print(appended_data)
