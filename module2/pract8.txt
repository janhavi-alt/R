Conducting Chi-square tests using PROC FREQ (SAS), Crosstabs with ChiSquare (SPSS), and chisq.test() (R).

install.packages("stats")
library(stats)
gender <- c("Male", "Female", "Female", "Male", "Male", "Female", "Female", "Male")
preference <- c("Yes", "No", "Yes", "Yes", "No", "Yes", "No", "No")
table_data <- table(gender, preference)
chi_result <- chisq.test(table_data)
print(table_data)
print(chi_result)
